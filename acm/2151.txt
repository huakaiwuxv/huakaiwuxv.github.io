#include<iostream>
#include<algorithm>  
#include<string>
using namespace std;
int N,P,M,T;
int dp[101][101];
//思路：状态方程:dp[i][j]=dp[i-1][j-1]+dp[i-1][j+1]  
int main()  
{  //2151
	while (scanf("%d%d%d%d",&N,&P,&M,&T)!=EOF)
	{
		memset(dp,0,sizeof(dp));
		dp[0][P]=1;
		int j;
		for (int i = 1; i <= M; i++)
		{
			for (int j = 1; j <= N; j++)
			{
				if(j==1)
				{
					dp[i][j]=dp[i-1][j+1];
				}
				else if(j==N)
				{
					dp[i][j]=dp[i-1][j-1];
				}
				else
					dp[i][j]=dp[i-1][j-1]+dp[i-1][j+1];
			}
		}
		cout<<dp[M][T]<<endl;
	}
	system("pause");
	return 0;
}
