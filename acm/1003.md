#include<iostream>
#include<algorithm>  
#include<string>
using namespace std;
int a[100001];
int b[100001];
int main()  
{  //1003
	int T,N;
	int first ,end;
	cin>>T;
	for (int i = 0; i < T; i++)
	{
		cin>>N;
		for (int j = 0; j < N; j++)
		{
			cin>>a[j];
		}
		b[0]=a[0];end=0;
		for (int j = 1; j < N; j++)
		{
			b[j]=max(a[j],b[j-1]+a[j]);
		}
		int m=-1003;
		for (int j = 0; j < N; j++)
		{
			if (m<b[j])
			{
				m=b[j];
				end=j;
			}
		}
		int n=m;
		for (int j = end; j >=0; j--)
		{
			n=n-a[j];
			if(n==0){
				first=j;
				break;
			}
		}
		cout<<"Case "<<i+1<<":"<<endl;
		cout<<m<<" "<<first+1<<" "<<end+1<<endl;
		if(i!=T-1)
			cout<<endl;
	}
	system("pause");
	return 0;
}
